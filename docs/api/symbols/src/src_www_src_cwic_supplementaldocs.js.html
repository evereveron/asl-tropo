 91 /**
 92  * @name call
 93  * @namespace
 94  * @property {Number} callId Unique identifier of the call.
 95  * @property {Boolean} exists <tt>true</tt> if the call exists.
 96  * @property {Boolean} audioMuted Audio is muted.
 97  * @property {Boolean} videoMuted Video is muted.
 98  * @property {String} callState Current state of the call.  May be one of the following:<ul>
 99  * <li>"OffHook" - receiver lifted in CTI control mode, or preparing to make a call in either mode.</li>
100  * <li>"OnHook" - call is ended or about to go OffHook. If !capabilities.canOriginateCall, then call is ended.</li>
101  * <li>"Ringout" - remote party is ringing.</li>
102  * <li>"Ringin" - incoming call.</li>
103  * <li>"Proceed" - if already on a call in CTI mode and there's an incoming call it may be in "Proceed" state.</li>
104  * <li>"Connected" - call is connected.</li>
105  * <li>"Hold" - call is held.</li>
106  * <li>"RemHold" - call is held on a shared-line device.</li>
107  * <li>"Resume" - n/a.</li>
108  * <li>"Busy" - remote line busy.</li>
109  * <li>"Reorder" - call failed.</li>
110  * <li>"Conference"</li>
111  * <li>"Dialing" - dialing number.</li>
112  * <li>"RemInUse" - call on a shared-line device.</li>
113  * <li>"HoldRevert"</li>
114  * <li>"Whisper"</li>
115  * <li>"Parked"</li>
116  * <li>"ParkRevert"</li>
117  * <li>"ParkRetrieved"</li>
118  * <li>"Preservation" - call is in preservation mode.</li>
119  * <li>"WaitingForDigits"</li
120  * ><li>"Spoof_Ringout"</li></ul>
121  * @property {String} callType One of the following: <ul><li>"Placed"</li><li>"Received"</li><li>"Missed"</li>
122  * @property {Date} start The <tt>new Date()</tt> when the conversationStart.cwic event was fired for the call.
123  * @property {String} videoDirection One of the following: <ul><li>"Inactive"</li><li>"SendRecv"</li><li>"SendOnly"</li><li>"RecvOnly"</li></ul>
124  * @property {Object} videoResolution Contains video resolution width and height values (in pixels). Included in the Call object any time the resolution changes.
125  * @property {Boolean} isConference <tt>true</tt> if the call is a conference call.
126  * @property {callcapability[]} capabilities A map that has capability name as the key and <tt>true</tt> as the value. 
127  * Call capabilities indicate operations you can perform on the call, and they should be used to control enabling/disabling/state/access to parts of the UI. Important capabilities are:
128  * <ul>
129  * <li>canOriginateCall - can create new call</li>
130  * <li>canAnswerCall - call can be answered</li>
131  * <li>canHold - call can be Held</li>
132  * <li>canResume - call can be Resumed</li>
133  * <li>canEndCall - call can be Ended</li>
134  * <li>canSendDigit - can send a DTMF digit to the call</li>
135  * <li>canDirectTransfer - can tranfer call</li>
136  * <li>canJoinAcrossLine - can join this call with another call on the same line with this capability also set</li>
137  * <li>canImmediateDivert - can iDivert (e.g. to voicemail)</li>
138  * <li>canUpdateVideoCapability - can update the videoDirection on the call. See {@link $.fn.cwic-updateConversation} examples.</li>
139  * <li>canMuteAudio - can mute audio on the call</li>
140  * <li>canUnmuteAudio - can unmute audio on the call</li>
141  * <li>canMuteVideo - can mute video on the call</li>
142  * <li>canUnmuteVideo - can unmute video on the call</li>
143  * </ul>
144  * @property {participant[]} participants Array of all remote participants on the call.
145  * @property {participant} participant Remote participant (first entry in participants list).
146  * @property {participant} localParticipant Local participant (your line info).
147  */
